--check table exist or not (dynamic sql)

CREATE OR REPLACE PROCEDURE SP_TABLES(P_TABLE VARCHAR2)AS
V_STR VARCHAR2(200);
V_CNT INT;
BEGIN
V_STR := 'SELECT COUNT(*)FROM USER_TABLES WHERE TABLE_NAME='||''''||P_TABLE||'''';
DBMS_OUTPUT.PUT_LINE(V_STR);
EXECUTE IMMEDIATE V_STR INTO V_CNT;
DBMS_OUTPUT.PUT_LINE(V_CNT);
END;

CREATE OR REPLACE PROCEDURE SP_COLS(P_TAB VARCHAR2)AS
V_STR VARCHAR2(200);
V_CNT INT;
BEGIN
V_STR := 'SELECT COUNT(*)FROM USER_TAB_COLUMNS WHERE TABLE_NAME='||''''||P_TAB||'''';
DBMS_OUTPUT.PUT_LINE(V_STR);
EXECUTE IMMEDIATE V_STR INTO V_CNT;
DBMS_OUTPUT.PUT_LINE(V_CNT);
END;
